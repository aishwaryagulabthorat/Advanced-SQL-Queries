*Solution - 1:*

```sql
with cte as 
(
select *,
	case when extract(month from login_timestamp) = 1 and  extract(year from login_timestamp) = 2024 then 'jan2024'
    when extract(month from login_timestamp) = 11 and  extract(year from login_timestamp) = 2023 then 'nov2023' 
    else 'no' end as 'status'
from logins)

select distinct c.user_id,u.user_name from cte c
inner join users u on c.user_id=u.user_id
where c.`status`='jan2024' and c.user_id not in (select user_id from cte where `status`='nov2023');
```

*Solution - 2:*

```sql
SELECT DISTINCT USER_ID
FROM logins
WHERE extract(year from login_timestamp) = '2024'
	AND extract(month from login_timestamp) = '1'

and user_id not in

(SELECT DISTINCT USER_ID
FROM logins
WHERE extract(year from login_timestamp) = '2023'
	AND extract(month from login_timestamp) = '11');
```
